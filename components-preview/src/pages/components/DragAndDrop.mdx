import { Playground } from 'docz'
import { DragAndDrop, ButtonBasic } from 'neutrino-design';
import 'neutrino-design/dist/index.css';

# Drag and Drop zone

<Playground>
  <div className="display-container margin">
    <DragAndDrop
      showOverlay={true}
      theme="dark"
      onDrop={data => console.log(data)}
      renderDropped={(ref, items) => {
        const droppedItems = [];
        if (items.files) for (let i = 0; i < items.files.length; i++) {
          droppedItems.push(items.files[i].name)
        }
        return droppedItems.map(i => <div key={i}>{i}</div>)
      }}
    >
      <ButtonBasic className="button">
        This is a <u>basic</u> button
      </ButtonBasic>
    </DragAndDrop>
  </div>
</Playground>
